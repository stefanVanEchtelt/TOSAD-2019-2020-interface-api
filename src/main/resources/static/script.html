CREATE OR REPLACE TRIGGER cijfer_klantje_trigger 
BEFORE UPDATE/INSERT 
ON klantje 
BEGIN 
           IF (1 > klantje.cijfer) or (1 = klantje.cijfer) 
                       raise error; 
           END IF; 
           EXCEPTION WHEN error THEN 
                       RAISE_APPLICATION_ERROR(-20002, 'een dikke error'); 
           END; 
END;
